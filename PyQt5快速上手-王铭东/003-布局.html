дёҖгҖҒQBoxLayout

зӣҙиҜ‘дёәпјҡзӣ’еӯҗеёғеұҖ

дёҖиҲ¬дҪҝз”Ёе®ғзҡ„дёӨдёӘеӯҗзұ»QHBoxLayout е’Ң QVBoxLayout иҙҹиҙЈж°ҙе№іе’ҢеһӮзӣҙеёғеұҖ

еһӮзӣҙеёғеұҖзӨәдҫӢпјҡ

import sys
from PyQt5.QtWidgets import QApplication, QVBoxLayout, QWidget, QPushButton, QGroupBox, QMainWindow
from PyQt5.QtCore import Qt


class MyWindow(QWidget):
    def __init__(self):
        # еҲҮи®°дёҖе®ҡиҰҒи°ғз”ЁзҲ¶зұ»зҡ„__init__ж–№жі•пјҢеӣ дёәе®ғйҮҢйқўжңүеҫҲеӨҡеҜ№UIз©әй—ҙзҡ„еҲқе§ӢеҢ–ж“ҚдҪң
        super().__init__()

        # и®ҫзҪ®еӨ§е°Ҹ
        self.resize(300, 300)
        # и®ҫзҪ®ж Үйўҳ
        self.setWindowTitle("еһӮзӣҙеёғеұҖ")

        # еһӮзӣҙеёғеұҖ
        layout = QVBoxLayout()

        # дҪңз”ЁжҳҜеңЁеёғеұҖеҷЁдёӯеўһеҠ дёҖдёӘдјёзј©йҮҸпјҢйҮҢйқўзҡ„еҸӮж•°иЎЁзӨәQSpacerItemзҡ„дёӘж•°пјҢй»ҳи®ӨеҖјдёәйӣ¶
        # дјҡе°ҶдҪ ж”ҫеңЁlayoutдёӯзҡ„з©әй—ҙеҺӢзј©жҲҗй»ҳи®Өзҡ„еӨ§е°Ҹ
        # дёӢйқўзҡ„з¬”иҜ•1пјҡ1пјҡ1пјҡ2
        layout.addStretch(1)

        # жҢүй’®1
        btn1 = QPushButton("жҢүй’®1")
        # ж·»еҠ еҲ°еёғеұҖеҷЁдёӯ
        # layout.addWidget(btn1, Qt.AlignmentFlag.AlignTop)
        layout.addWidget(btn1)

        layout.addStretch(1)

        # жҢүй’®2
        btn2 = QPushButton("жҢүй’®2")
        # ж·»еҠ еҲ°еёғеұҖеҷЁ
        layout.addWidget(btn2)

        layout.addStretch(1)

        # жҢүй’®3
        btn3 = QPushButton("жҢүй’®3")
        # ж·»еҠ еҲ°еёғеұҖеҷЁ
        layout.addWidget(btn3)

        layout.addStretch(2)

        self.setLayout(layout)


if __name__ == '__main__':
    app = QApplication(sys.argv)

    # еҲӣе»әдёҖдёӘQWidgetеӯҗзұ»
    w = MyWindow()
    w.show()

    app.exec()

import sys
from PyQt5.QtWidgets import QApplication, QWidget, QGroupBox, QVBoxLayout, QHBoxLayout, QRadioButton


class MyWindow(QWidget):

    def __init__(self):
        super().__init__()
        self.init_ui()

    def init_ui(self):
        # жңҖеӨ–еұӮзҡ„еһӮзӣҙеёғеұҖпјҢеҢ…еҗ«дёӨйғЁеҲҶпјҡзҲұеҘҪе’ҢжҖ§еҲ«
        container = QVBoxLayout()

        # -----еҲӣе»әз¬¬1дёӘз»„пјҢж·»еҠ еӨҡдёӘз»„д»¶-----
        # hobby дё»иҰҒжҳҜдҝқиҜҒд»–д»¬жҳҜдёҖдёӘз»„гҖӮ
        hobby_box = QGroupBox("зҲұеҘҪ")
        # v_layout дҝқиҜҒдёүдёӘзҲұеҘҪжҳҜеһӮзӣҙж‘Ҷж”ҫ
        v_layout = QVBoxLayout()
        btn1 = QRadioButton("жҠҪзғҹ")
        btn2 = QRadioButton("е–қй…’")
        btn3 = QRadioButton("зғ«еӨҙ")
        # ж·»еҠ еҲ°v_layoutдёӯ
        v_layout.addWidget(btn1)
        v_layout.addWidget(btn2)
        v_layout.addWidget(btn3)
        # жҠҠv_layoutж·»еҠ еҲ°hobby_boxдёӯ
        hobby_box.setLayout(v_layout)

        # -----еҲӣе»әз¬¬2дёӘз»„пјҢж·»еҠ еӨҡдёӘз»„д»¶-----
        # жҖ§еҲ«з»„
        gender_box = QGroupBox("жҖ§еҲ«")
        # жҖ§еҲ«е®№еҷЁ
        h_layout = QHBoxLayout()
        # жҖ§еҲ«йҖүйЎ№
        btn4 = QRadioButton("з”·")
        btn5 = QRadioButton("еҘі")
        # иҝҪеҠ еҲ°жҖ§еҲ«е®№еҷЁдёӯ
        h_layout.addWidget(btn4)
        h_layout.addWidget(btn5)
        # ж·»еҠ еҲ° boxдёӯ
        gender_box.setLayout(h_layout)

        # жҠҠзҲұеҘҪзҡ„еҶ…е®№ж·»еҠ еҲ°е®№еҷЁдёӯ
        container.addWidget(hobby_box)
        # жҠҠжҖ§еҲ«зҡ„еҶ…е®№ж·»еҠ еҲ°е®№еҷЁдёӯ
        container.addWidget(gender_box)

        # и®ҫзҪ®зӘ—еҸЈжҳҫзӨәзҡ„еҶ…е®№жҳҜжңҖеӨ–еұӮе®№еҷЁ
        self.setLayout(container)


if __name__ == '__main__':
    app = QApplication(sys.argv)

    w = MyWindow()
    w.show()

    app.exec()

иҝҗиЎҢж•Ҳжһңпјҡ

жіЁж„Ҹ

  • йҖҡиҝҮдёҠйқўзҡ„зӨәдҫӢпјҢжҲ‘д»¬иғҪеӨҹзңӢеҲ°ж°ҙе№іеёғеұҖеҷЁдёҺеһӮзӣҙеёғеұҖеҷЁжҳҜеҸҜд»Ҙж··еҗҲдҪҝз”ЁеҚіеөҢеҘ—дҪҝз”Ё

дәҢгҖҒQGridLayout

зҪ‘ж јеёғеұҖпјҢжңүзҡ„дәәз§°д№Ӣдёәд№қе®«ж јеёғеұҖ

import sys
from PyQt5.QtWidgets import QApplication, QWidget, QVBoxLayout, QPushButton, QLineEdit, QGridLayout


class MyWindow(QWidget):

    def __init__(self):
        super().__init__()
        self.init_ui()

    def init_ui(self):
        self.setWindowTitle("и®Ўз®—еҷЁ")

        # еҮҶеӨҮж•°жҚ®
        data = {
            0: ["7", "8", "9", "+", "("],
            1: ["4", "5", "6", "-", ")"],
            2: ["1", "2", "3", "*", "<-"],
            3: ["0", ".", "=", "/", "C"]
        }

        # ж•ҙдҪ“еһӮзӣҙеёғеұҖ
        layout = QVBoxLayout()

        # иҫ“е…ҘжЎҶ
        edit = QLineEdit()
        edit.setPlaceholderText("иҜ·иҫ“е…ҘеҶ…е®№")
        # жҠҠиҫ“е…ҘжЎҶж·»еҠ еҲ°е®№еҷЁдёӯ
        layout.addWidget(edit)

        # зҪ‘ж јеёғеұҖ
        grid = QGridLayout()

        # еҫӘзҺҜеҲӣе»әиҝҪеҠ иҝӣеҺ»
        for line_number, line_data in data.items():
            # жӯӨж—¶line_numberжҳҜз¬¬еҮ иЎҢпјҢline_dataжҳҜеҪ“еүҚиЎҢзҡ„ж•°жҚ®
            for col_number, number in enumerate(line_data):
                # жӯӨж—¶col_numberжҳҜз¬¬еҮ еҲ—пјҢnumberжҳҜиҰҒжҳҫзӨәзҡ„з¬ҰеҸ·
                btn = QPushButton(number)
                # grid.addWidget(btn)
                grid.addWidget(btn, line_number, col_number)

        # жҠҠзҪ‘ж јеёғеұҖиҝҪеҠ еҲ°е®№еҷЁдёӯ
        layout.addLayout(grid)

        self.setLayout(layout)


if __name__ == '__main__':
    app = QApplication(sys.argv)

    w = MyWindow()
    w.show()

    app.exec()
